156 __STATIC_INLINE void CLK_SysTickDelay(uint32_t us)
157 {
158  SysTick->LOAD = us * CyclesPerUs;
159  SysTick->VAL = (0x00);
160  SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk | SysTick_CTRL_ENABLE_Msk;
161 
162  /* Waiting for down-count to zero */
163  while((SysTick->CTRL & SysTick_CTRL_COUNTFLAG_Msk) == 0);
164 
165  /* Disable SysTick counter */
166  SysTick->CTRL = 0;
167 }
