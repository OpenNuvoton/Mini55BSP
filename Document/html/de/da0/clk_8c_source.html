336void CLK_EnableSysTick(uint32_t u32ClkSrc, uint32_t u32Count)
337{
338 /* Set System Tick counter disabled */

341 /* Set System Tick clock source */
342 if( u32ClkSrc == CLK_CLKSEL0_STCLKSEL_HCLK )
